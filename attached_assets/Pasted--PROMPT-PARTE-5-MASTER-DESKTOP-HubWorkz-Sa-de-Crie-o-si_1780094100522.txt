HOOK useAuth() — criar em src/hooks/useAuth.ts:
export function useAuth() {
  const [user, setUser] = useState<User | null>(null)
  const [profile, setProfile] = useState<Profile | null>(null)
  const [loading, setLoading] = useState(true)
  useEffect(() => {
    supabase.auth.getSession().then(({ data: { session } }) => {
      setUser(session?.user ?? null)
      if (session?.user) fetchProfile(session.user.id)
    })
  }, [])
  return { user, profile, role: profile?.role, loading }
}

PADRÃO DE COMPONENTE (copiar para cada página):
import { useEffect, useState } from 'react'
import { supabase } from '../lib/supabase'
import { useAuth } from '../hooks/useAuth'
import type { Paciente } from '@hubworkz/shared'

export function Pacientes() {
  const { profile } = useAuth()
  const [items, setItems] = useState<Paciente[]>([])

  useEffect(() => {
    supabase
      .from('pacientes')
      .select('*, medicamento:medicamentos(*)')
      .eq('clinica_id', profile?.clinica_id)
      .order('created_at', { ascending: false })
      .then(({ data }) => setItems(data || []))
  }, [profile])
  // ...
}
